가상 클래스

:link :visited :hover :active :before :after

:hover, :active, :link, :visited 는 a 태그와 함께 링크를 데코레이션 할 때 많이 사용된다. 다른 태그랑도 쓸 수는 있을듯

a:link {color: #ff00ff; text-decoration: none;} /*방문전 링크일때 분홍색으로

a:visited {color: #000000; text-decoration: none;} /*방문후 링크일때 검은색으로 보임

a:hover {color: #ff0000;} /*마우스 올렸을때만(마우스오버) 빨간색이고싶어

a:active {color: #00ff00; text-decoration: underline;} /*마우스 클릭했을 때..키를 눌렀을때 초록색 */

h2:before {content: "#";}
h2:after {content: "$";}
